This weeks data describes landlines and mobile phones, specifically number of subscriptions of each for each country in the world.

This would be a good time to use the waterfall chart I made the other day (in the data visualization project). You can see new subscriptions or reductions in subscriptions over time. Lets try it with the US for simplicity. Heres the original visualization I made, inspired by Anh Hoang Duc.

set.seed(123)
data <- data.frame(date = seq(1, 372, by = 31) %>% as_date)
data <- data %>% 
  mutate(new = abs(rnorm(12, 100, 10)) %>% round(0)) %>% 
  mutate(churn = abs(rnorm(12, 50, 30)) %>% round(0)) %>% 
  mutate(net = new - churn)  %>% 
  mutate(eop = cumsum(net)) %>% 
  select(-net)

waterfall_data <- data %>% 
  mutate(
    ymax = eop + churn
  ) %>% 
  pivot_longer(cols = new:churn) %>% 
  mutate(
    name = factor(name, levels = c('new','churn')), # Proper ordering
    xmin = case_when(
      name == 'new' ~ date - days(12),
      name == 'churn' ~ date
    ),
    xmax = case_when(
      name == 'new' ~ date,
      name == 'churn' ~ date + days(12)
    ),
    ymin = case_when(
      name == 'new' ~ ymax - value,
      name == 'churn' ~ eop
    )
  )

waterfall_data %>% 
  ggplot() + 
  geom_rect(aes(xmin = xmin, xmax = xmax, ymin = ymin, ymax = ymax,
            group = name, fill = name)) + 
  geom_line(aes(x = date, y = eop), size = 1, color = '#193e73') + 
  geom_point(aes(x = date, y = eop), size = 2, color = '#193e73') + 
  labs(
    title = "Waterfall Chart",
    subtitle = "Net change, accompanied by a decomposition of the net change"
  ) + 
  scale_fill_manual(values = c('#808080', '#ff5959')) + 
  theme_minimal() + 
  theme(
    axis.line.x = element_line(),
    axis.line.y = element_line()
  )
